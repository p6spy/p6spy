42  public class SpyDotPropertiesTest extends BaseTestCase {
43    private Properties systemProps;
44    private ClassLoader currentThreadLoader;
45    private File testFile;
46  
47    @Before
48    public void setup() {
49      systemProps = new Properties();
50      for( String key : System.getProperties().stringPropertyNames()) {
51        systemProps.setProperty(key, System.getProperty(key));
52      }
53      currentThreadLoader = Thread.currentThread().getContextClassLoader();
54    }
55    
56    @After
57    public void cleanup() throws IOException {
58      System.setProperties(systemProps);
59      Thread.currentThread().setContextClassLoader(currentThreadLoader);
60      if( testFile != null ) {
61        testFile.delete();
62      }
63      
64    }
65  
66    @Test
67    public void testLoadFromJarFile() throws Exception {
68      // create new classloader
69      ClassLoader tempLoader = new URLClassLoader(new URL[]{new File("src/test/resources/SpyDotPropertiesTest.jar").toURI().toURL()}, currentThreadLoader);
70      Thread.currentThread().setContextClassLoader(tempLoader);
71  
72      // configure the file to load
73      System.setProperty(SpyDotProperties.OPTIONS_FILE_PROPERTY, "SpyDotPropertiesTest.properties");
74      SpyDotProperties props = new SpyDotProperties();
75      assertNotNull("properties not loaded!", props.getOptions());
76    }
77  
78    @Test
79    public void testLoadFromCurrentWorkingDirectory() throws Exception {
80      // create the file to load
81      testFile = File.createTempFile("test", null, new File("."));
82      testFile.deleteOnExit();
83      PrintWriter pw = new PrintWriter(testFile);
84      pw.println("modulelist=testModule");
85      pw.close();
86      System.setProperty(SpyDotProperties.OPTIONS_FILE_PROPERTY, testFile.getName());
87  
88      SpyDotProperties props = new SpyDotProperties();
89      assertNotNull("properties not loaded!", props.getOptions());
90      
91      if( !testFile.delete() ) {      
92        // if the file was not closed properly, the delete will fail!
93        fail("temporary file was not deleted!");
94      }
95      
96    }
97  
98    @Test
99    public void testLoadFromP6Home() throws Exception {
100     
101     // create p6home
102     String p6home = "target/p6home";
103     System.setProperty("p6.home", p6home);
104     File p6HomeDir = new File(p6home);

116   }
