20  package com.p6spy.engine.common;
21  
22  import com.p6spy.engine.proxy.Delegate;
23  
24  import java.lang.reflect.Method;
25  import java.sql.SQLException;
26  import java.sql.Wrapper;
27  
28  /**
29   * Implementation of {@link java.sql.Wrapper#unwrap(Class)} for proxy classes.
30   */
31  public class P6WrapperUnwrapDelegate implements Delegate {
32    @Override
33    public Object invoke(final Object proxy, final Object underlying, final Method method, final Object[] args) throws Throwable {
34      Class<?> iface = (Class<?>) args[0];
35      Object result = null;
36  
37      // if the proxy directly implements the interface or extends it, return the proxy
38      if( iface.isAssignableFrom(proxy.getClass()) ) {
39        result = proxy;
40      }
41  
42      // if the proxied object directly implements the interface or extends it, return
43      // the proxied object
44      else if (iface.isAssignableFrom(underlying.getClass())) {
45        result = underlying;
46      }
47  
48      // if the proxied object implements the wrapper interface, then
49      // return the result of it's unwrap method.
50      else if (Wrapper.class.isAssignableFrom(underlying.getClass())) {
51        result = ((Wrapper) underlying).unwrap(iface);
52      }
53  
54      else {
55        /*
56           This line of code can only be reached when the underlying object does not implement the wrapper
57           interface.  This would mean that either the JDBC driver or the wrapper of the underlying object
58           does not implement the JDBC 4.0 API.
59         */
60        throw new SQLException("Can not unwrap to "+iface.getName());
61      }
62  
63      return iface.cast(result);
64    }
65  }
