63  // Description: Class file for detecting long-running statements that may indicate
64  //              a database outage problem.
65  
66  package com.p6spy.engine.outage;
67  
68  import com.p6spy.engine.common.*;
69  import java.util.concurrent.ConcurrentHashMap;
70  import java.util.concurrent.ConcurrentMap;
71  
72  /**
73   * This class is a singleton. Since P6Spy is normally loaded by the system
74   * classpath, it is a normally a singleton across the JVM.
75   * The instance will determine if a statement has been long running and log
76   * the statement when such a statment is found. It accomplishes this by
77   * spawning a daemon thread which will wake up at configurable intervals of
78   * time (defined in seconds) and check if a statement is still executing since
79   * the last time the thread was awake. This is accomplished by this instance
80   * maintaining a list of active statements. The P6Statement and P6PreparedStatement
81   * objects will reqister their call with the instance just before the SQL call
82   * is delegated to the real driver. Once that statement finishes executing, the
83   * statement object will unregister that call from list. If during that time
84   * the thread in this instance sees the execute time exceed the threshold, it is
85   * flagged as a long-running statement and logged. The statement will continue
86   * to be logged if it appears in future iterations of the sleep/wake cycle.
87   * This class is implemented with lazy thread sychronization. The list of
88   * pending statements is a synchronized container, the rest is left open to
89   * thread hazards to reduce the performance impact. So the logging results might be
90   * slightly unreliable, but we aren't dealing with bank accounts here so thats
91   * okay.
92   */
93  public class P6OutageDetector implements Runnable {
94      private ConcurrentMap<Object, InvocationInfo> pendingMessages;
95  
96      // flag that indicates that the thread should stop running
97      private boolean haltThread;
98  
99      // singleton contruct
100     private static P6OutageDetector instance;
101 
102     /** Creates new P6OutageDetector */
103     protected P6OutageDetector() {
104         pendingMessages = new ConcurrentHashMap<Object, InvocationInfo>();
105 
106         P6LogQuery.debug("P6Spy - P6OutageDetector has been invoked.");
107         P6LogQuery.debug("P6Spy - P6OutageOptions.getOutageDetectionIntervalMS() = " + P6OutageOptions.getOutageDetectionIntervalMS());
108     }
109 
110     /**
111      * Gets the instance of the detector. A side effect of the first call to this method is that the
112      * auxillary thread will be kicked off here.
113      *
114      * @return the P6OutageDetector instance
115      */
116     static public synchronized P6OutageDetector getInstance() {
117         if (instance == null) {
118             instance = new P6OutageDetector();
119 
120             // create and run the auxilliary thread
121             // make it a deamon thread so it won't prevent the server from
122             // shutting down when it wants to.
123             ThreadGroup group = new ThreadGroup("P6SpyThreadGroup");
124             group.setDaemon(true);
125             Thread outageThread = new Thread(group, instance, "P6SpyOutageThread");
126             outageThread.start();
127         }
128         return instance;
129     }
130 
131     /**
132      * Method for running the auxillary thread.
133      */
134     public void run() {
135         while (!haltThread) {
136             detectOutage();
137 
138             try {
139                 // sleep for the configured interval
140                 // don't cache this value since the props file may be reloaded
141                 // and this value might change
142                 Thread.sleep(P6OutageOptions.getOutageDetectionIntervalMS());
143             } catch (Exception e) {
144             }
145         }
146     }
147 
148     /**
149      * Tells the auxillary thread to stop executing. Thread will exit upon waking next.
150      */
151     public void shutdown() {
152         haltThread = true;
153     }
154 
155     /**
156      * Registers the execution of a statement. This should be called just before the statement is
157      * passed to the real driver.
158      */
159     public void registerInvocation(Object jdbcObject, long startTime, String category, String ps, String sql) {
160         pendingMessages.put(jdbcObject, new InvocationInfo(startTime, category, ps, sql));
161     }
162 
163     /**
164      * Unregisters the execution of a statement. This should be called just after the statement is
165      * passed to the real driver.
166      */
167     public void unregisterInvocation(Object jdbcObject) {
168         pendingMessages.remove(jdbcObject);
169     }
170 
171     private void detectOutage() {
172         int listSize = pendingMessages.size();
173         if (listSize == 0) {
174             return;
175         }
176 
177         P6LogQuery.debug("P6Spy - detectOutage.pendingMessage.size = " + listSize);
178 
179         long currentTime = System.currentTimeMillis();
180         long threshold = P6OutageOptions.getOutageDetectionIntervalMS();
181 
182         for (Object jdbcObject : pendingMessages.keySet()) {
183             // here is a thread hazard that we'll be lazy about. Another thread
184             // might have already removed the entry from the messages map, so we
185             // need to check if the result is null
186             InvocationInfo ii = pendingMessages.get(jdbcObject);
187             if (ii == null) {
188                 continue;
189             }
190 
191             // has this statement exceeded the threshold?
192             if ((currentTime - ii.startTime) > threshold) {
193                 P6LogQuery.debug("P6Spy - statement exceeded threshold - check log.");
194                 logOutage(ii);
195             }
196         }
197     }
198 
199     private void logOutage(InvocationInfo ii) {
200         P6LogQuery.logElapsed(-1, ii.startTime, "OUTAGE", ii.preparedStmt, ii.sql);
201     }
202 
203 }

205 // inner class to hold the info about a specific statement invocation
206 class InvocationInfo {
207     public long startTime;
208 
209     public String category;
210 
211     public String preparedStmt;
212 
213     public String sql;
214 
215     public InvocationInfo(long startTime, String category, String ps, String sql) {
216         this.startTime = startTime;
217         this.category = category;

219         this.sql = sql;
220     }
