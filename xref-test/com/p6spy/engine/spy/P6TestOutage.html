62    @Test
63    public void testOutage() throws SQLException {
64      // exec fast query => no outage detected
65      callSleep(1);
66      Assert.assertFalse(super.getLastLogEntry().contains(Category.OUTAGE.toString()));
67      Assert.assertTrue(super.getLastLogEntry().contains("CALL SLEEP"));
68  
69      // exec slooooow query => outage detected
70      callSleep(3000);
71      Assert.assertTrue(super.getLastButOneLogEntry().contains(Category.OUTAGE.toString()));
72      Assert.assertTrue(super.getLastLogEntry().contains("CALL SLEEP"));
73    }

75    @Test
76    public void testCallingSetMethodsOnStatementInterface() throws SQLException {
77  
78      // not a great test - just protecting against regression for issue #275
79  
80      String sql = "select * from customers where id = ?";
81      PreparedStatement prep = connection.prepareStatement(sql);
82  
83      prep.setMaxRows(1);
84      assertEquals(1, prep.getMaxRows());
85  
86      prep.close();
87    }
88  
89    private void callSleep(long miliseconds) throws SQLException {
90      final Statement statement = connection.createStatement();
91      statement.executeQuery("CALL SLEEP( " + miliseconds + ")");
92      statement.close();
93    }
