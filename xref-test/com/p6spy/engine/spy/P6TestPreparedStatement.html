117   @Test
118   public void testSameColumnNameInMultipleTables() throws SQLException {
119     try {
120         // insert test data
121         {
122           final String update = "insert into prepstmt_test values (?, ?)";
123           final PreparedStatement prep = getPreparedStatement(update);
124           prep.setString(1, "prepstmt_test_col1");
125           prep.setInt(2, 1);
126           prep.executeUpdate();
127           prep.close();
128         }
129         {
130           final String update = "insert into prepstmt_test2 values (?, ?)";
131           final PreparedStatement prep = getPreparedStatement(update);
132           prep.setString(1, "prepstmt_test_col2");
133           prep.setInt(2, 1);
134           prep.executeUpdate();
135           prep.close();
136         }
137 
138         super.clearLogEntries();
139 
140         // let's check that returned data are reported correctly
141         // => don't filter 'result' and 'resultset'
142         P6LogOptions.getActiveInstance().setExcludecategories("");
143 
144         final String query = "select prepstmt_test.col1, prepstmt_test2.col1, prepstmt_test.col2, prepstmt_test2.col2 from prepstmt_test, prepstmt_test2 where prepstmt_test.col2 = prepstmt_test2.col2 and prepstmt_test.col1 = ? and prepstmt_test2.col1 = ?";
145         final PreparedStatement prep = getPreparedStatement(query);
146         prep.setString(1, "prepstmt_test_col1");
147         prep.setString(2, "prepstmt_test_col2");
148         final ResultSet rs = prep.executeQuery();
149 
150         // check "statement" logged properly
151         assertNotNull(super.getLastLogEntry());
152         assertTrue("prepared statement not logged properly",
153             super.getLastLogEntry().contains("statement"));
154         assertTrue("prepared statement not logged properly", super.getLastLogEntry().contains(query));
155         assertTrue(
156             "prepared statement not logged properly",
157             super.getLastLogEntry().contains(
158                 query.replaceFirst("\\?", "\'prepstmt_test_col1\'").replaceFirst("\\?",
159                     "\'prepstmt_test_col2\'")));
160 
161         // check returned (real) data not messed up
162         while (rs.next()) {
163           assertEquals("returned values messed up", "prepstmt_test_col1", rs.getString(1));
164           assertEquals("returned values messed up", "prepstmt_test_col2", rs.getString(2));
165           assertEquals("returned values messed up", 1, rs.getInt(3));
166           assertEquals("returned values messed up", 1, rs.getInt(4));
167         }
168         rs.close();
169         prep.close();
170 
171         // check "resultset" logged properly
172         assertNotNull(super.getLastLogEntry());
173         assertTrue("resultset not logged", super.getLastLogEntry().contains("resultset"));
174         assertTrue(
175             "logged resultset holds incorrect values",
176             super.getLastLogEntry().endsWith(
177                 "1 = prepstmt_test_col1, 2 = prepstmt_test_col2, 3 = 1, 4 = 1"));
178 
179         P6LogOptions.getActiveInstance().setExcludecategories("result,resultset");
180 
181     } catch (Exception e) {
182       fail(e.getMessage() + " due to error:\n" + getStackTrace(e));
183     }

186   @Test
187   public void testExecuteUpdate() {
188     try {
189       // test a basic insert
190       String update = "insert into prepstmt_test values (?, ?)";
191       PreparedStatement prep = getPreparedStatement(update);
192       prep.setString(1, "miller");
193       prep.setInt(2, 1);
194       prep.executeUpdate();
195       assertTrue(super.getLastLogEntry().contains(update));
196       assertTrue(super.getLastLogEntry().contains("miller"));
197       assertTrue(super.getLastLogEntry().contains("1"));
198 
199 
200       // test dynamic allocation of P6_MAX_FIELDS
201       int MaxFields = 10;
202       StringBuffer bigSelect = new StringBuffer(MaxFields);
203       bigSelect.append("select count(*) from prepstmt_test where");
204       for (int i = 0; i < MaxFields; i++) {
205         if (i > 0) {
206           bigSelect.append(" or ");
207         }
208         bigSelect.append(" col2=?");
209       }
210       prep.close();
211 
212       prep = getPreparedStatement(bigSelect.toString());
213       for (int i = 1; i <= MaxFields; i++) {
214         prep.setInt(i, i);
215       }
216       prep.close();
217 
218       // test batch inserts
219       update = "insert into prepstmt_test values (?,?)";
220       prep = getPreparedStatement(update);
221       prep.setString(1, "danny");
222       prep.setInt(2, 2);
223       prep.addBatch();
224       prep.setString(1, "denver");
225       prep.setInt(2, 3);
226       prep.addBatch();
227       prep.setString(1, "aspen");
228       prep.setInt(2, 4);
229       prep.addBatch();
230       prep.executeBatch();
231       assertTrue(super.getLastLogEntry().contains(update));
232       assertTrue(super.getLastLogEntry().contains("aspen"));
233       assertTrue(super.getLastLogEntry().contains("4"));
234       prep.close();
235 
236       String query = "select count(*) from prepstmt_test";
237       prep = getPreparedStatement(query);
238       ResultSet rs = prep.executeQuery();
239       rs.next();
240       assertEquals(4, rs.getInt(1));
241 
242       rs.close();
243       prep.close();
244     } catch (Exception e) {
245       fail(e.getMessage() + " due to error:\n" + getStackTrace(e));
246     }
