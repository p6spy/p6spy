107   @Test
108   public void testExecutionThreshold() throws SQLException {
109     Statement statement = connection.createStatement();
110 
111     try {
112       // set the execution threshold very low
113       P6LogOptions.getActiveInstance().setExecutionThreshold("0");
114 
115       // test a basic select
116       String query = "select count(*) from customers";
117       ResultSet rs = statement.executeQuery(query);
118       assertTrue(super.getLastLogEntry().contains(query));
119       // finally just make sure the query executed!
120       rs.next();
121       assertTrue(rs.getInt(1) > 0);
122       rs.close();
123 
124       // now increase the execution threshold and make sure the query is not captured
125       P6LogOptions.getActiveInstance().setExecutionThreshold("10000");
126 
127       // test a basic select
128       String nextQuery = "select count(*) from customers where 1 = 2";
129       rs = statement.executeQuery(nextQuery);
130       // make sure the previous query is still the last query
131       assertTrue(super.getLastLogEntry().contains(query));
132       // and of course that the new query isn't
133       assertFalse(super.getLastLogEntry().contains(nextQuery));
134       // finally just make sure the query executed!
135       rs.next();
136       assertEquals(0, rs.getInt(1));
137       rs.close();
138 
139       P6LogOptions.getActiveInstance().setExecutionThreshold("0");
140 
141       // finally, just make sure it now works as expected
142       rs = statement.executeQuery(nextQuery);
143       assertTrue(super.getLastLogEntry().contains(nextQuery));
144       rs.next();
145       assertEquals(0, rs.getInt(1));
146       rs.close();
147     } finally {
148       if (statement != null) {
149         statement.close();
150       }
151     }
152   }
