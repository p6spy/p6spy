93    @Test
94    public void testSameColumnNameInMultipleTables() throws SQLException {
95      try {
96          // insert test data
97          {
98            final String update = "insert into prepstmt_test values (?, ?)";
99            final PreparedStatement prep = getPreparedStatement(update);
100           prep.setString(1, "prepstmt_test_col1");
101           prep.setInt(2, 1);
102           prep.executeUpdate();
103           prep.close();
104         }
105         {
106           final String update = "insert into prepstmt_test2 values (?, ?)";
107           final PreparedStatement prep = getPreparedStatement(update);
108           prep.setString(1, "prepstmt_test_col2");
109           prep.setInt(2, 1);
110           prep.executeUpdate();
111           prep.close();
112         }
113   
114         super.clearLogEnties();
115   
116         // let's check that returned data are reported correctly
117         P6LogOptions.getActiveInstance().setExcludecategories("-result,-resultset");
118   
119         final String query = "select prepstmt_test.col1, prepstmt_test2.col1, prepstmt_test.col2, prepstmt_test2.col2 from prepstmt_test, prepstmt_test2 where prepstmt_test.col2 = prepstmt_test2.col2 and prepstmt_test.col1 = ? and prepstmt_test2.col1 = ?";
120         final PreparedStatement prep = getPreparedStatement(query);
121         prep.setString(1, "prepstmt_test_col1");
122         prep.setString(2, "prepstmt_test_col2");
123         final ResultSet rs = prep.executeQuery();
124   
125         // check "statement" logged properly
126         assertNotNull(super.getLastLogEntry());
127         assertTrue("prepared statement not logged properly",
128             super.getLastLogEntry().contains("statement"));
129         assertTrue("prepared statement not logged properly", super.getLastLogEntry().contains(query));
130         assertTrue(
131             "prepared statement not logged properly",
132             super.getLastLogEntry().contains(
133                 query.replaceFirst("\\?", "\'prepstmt_test_col1\'").replaceFirst("\\?",
134                     "\'prepstmt_test_col2\'")));
135   
136         // check returned (real) data not messed up
137         while (rs.next()) {
138           assertEquals("returned values messed up", "prepstmt_test_col1", rs.getString(1));
139           assertEquals("returned values messed up", "prepstmt_test_col2", rs.getString(2));
140           assertEquals("returned values messed up", 1, rs.getInt(3));
141           assertEquals("returned values messed up", 1, rs.getInt(4));
142         }
143         rs.close();
144         prep.close();
145   
146         // check "resultset" logged properly
147         assertNotNull(super.getLastLogEntry());
148         assertTrue("resultset not logged", super.getLastLogEntry().contains("resultset"));
149         assertTrue(
150             "logged resultset holds incorrect values",
151             super.getLastLogEntry().endsWith(
152                 "1 = prepstmt_test_col1, 2 = prepstmt_test_col2, 3 = 1, 4 = 1"));
153   
154         P6LogOptions.getActiveInstance().setExcludecategories("result,resultset");
155       
156     } catch (Exception e) {
157       fail(e.getMessage() + " due to error: " + getStackTrace(e));
158     }

161   @Test
162   public void testExecuteUpdate() {
163     try {
164       // test a basic insert
165       String update = "insert into prepstmt_test values (?, ?)";
166       PreparedStatement prep = getPreparedStatement(update);
167       prep.setString(1, "miller");
168       prep.setInt(2, 1);
169       prep.executeUpdate();
170       assertTrue(super.getLastLogEntry().contains(update));
171       assertTrue(super.getLastLogEntry().contains("miller"));
172       assertTrue(super.getLastLogEntry().contains("1"));
173 
174 
175       // test dynamic allocation of P6_MAX_FIELDS
176       int MaxFields = 10;
177       StringBuffer bigSelect = new StringBuffer(MaxFields);
178       bigSelect.append("select count(*) from prepstmt_test where");
179       for (int i = 0; i < MaxFields; i++) {
180         if (i > 0) {
181           bigSelect.append(" or ");
182         }
183         bigSelect.append(" col2=?");
184       }
185       prep.close();
186       
187       prep = getPreparedStatement(bigSelect.toString());
188       for (int i = 1; i <= MaxFields; i++) {
189         prep.setInt(i, i);
190       }
191       prep.close();
192       
193       // test batch inserts
194       update = "insert into prepstmt_test values (?,?)";
195       prep = getPreparedStatement(update);
196       prep.setString(1, "danny");
197       prep.setInt(2, 2);
198       prep.addBatch();
199       prep.setString(1, "denver");
200       prep.setInt(2, 3);
201       prep.addBatch();
202       prep.setString(1, "aspen");
203       prep.setInt(2, 4);
204       prep.addBatch();
205       prep.executeBatch();
206       assertTrue(super.getLastLogEntry().contains(update));
207       assertTrue(super.getLastLogEntry().contains("aspen"));
208       assertTrue(super.getLastLogEntry().contains("4"));
209       prep.close();
210       
211       String query = "select count(*) from prepstmt_test";
212       prep = getPreparedStatement(query);
213       ResultSet rs = prep.executeQuery();
214       rs.next();
215       assertEquals(4, rs.getInt(1));
216       
217       rs.close();
218       prep.close();
219     } catch (Exception e) {
220       fail(e.getMessage() + " due to error: " + getStackTrace(e));
221     }
222   }
