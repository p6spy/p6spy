161     protected void bindDataSource() throws SQLException {
162         // can be set when object is bound to JDNI, or
163         // can be loaded from spy.properties
164         if (rdsName == null) {
165             rdsName = P6SpyOptions.getRealDataSource();
166         }
167         if (rdsName == null) {
168 
169             throw new SQLException("P6DataSource: no value for Real Data Source Name, cannot perform jndi lookup");
170         }
171         try {
172             Hashtable env = null;
173             String factory;
174             if ((factory = P6SpyOptions.getJNDIContextFactory()) != null) {
175                 env = new Hashtable();
176                 env.put(Context.INITIAL_CONTEXT_FACTORY, factory);
177                 String url = P6SpyOptions.getJNDIContextProviderURL();
178                 if (url != null) {
179                     env.put(Context.PROVIDER_URL, url);
180                 }
181                 String custom = P6SpyOptions.getJNDIContextCustom();
182                 if (custom != null) {
183                     StringTokenizer st = new StringTokenizer(custom, ",", false);
184                     while (st.hasMoreElements()) {
185                         String pair = st.nextToken();
186                         StringTokenizer pst = new StringTokenizer(pair, ";", false);
187                         if (pst.hasMoreElements()) {
188                             String name = pst.nextToken();
189                             if (pst.hasMoreElements()) {
190                                 String value = pst.nextToken();
191                                 env.put(name, value);
192                             }
193                         }
194                     }
195                 }
196             }
197             InitialContext ctx;
198             if (env != null) {
199                 ctx = new InitialContext(env);
200             } else {
201                 ctx = new InitialContext();
202             }
203             rds = (DataSource) ctx.lookup(rdsName);
204 
205             // Set any properties that the spy.properties file contains
206             // that are supported by set methods in this class
207 
208             String dsProps = P6SpyOptions.getRealDataSourceProperties();
209 
210             if (dsProps != null) {
211                 Hashtable props = null;
212 
213                 StringTokenizer st = new StringTokenizer(dsProps, ",", false);
214                 while (st.hasMoreElements()) {
215                     String pair = st.nextToken();
216                     StringTokenizer pst = new StringTokenizer(pair, ";", false);
217                     if (pst.hasMoreElements()) {
218                         String name = pst.nextToken();
219                         if (pst.hasMoreElements()) {
220                             String value = pst.nextToken();
221                             if (props == null) {
222                                 props = new Hashtable();
223                             }
224                             props.put(name, value);
225                         }
226                     }
227                 }
228                 Hashtable matchedProps = new Hashtable();
229                 if (props != null) {
230                     Class klass = rds.getClass();
231 
232                     // find the setter methods in the class, and
233                     // see if the datasource properties collected
234                     // from the spy.properties file contains any matching
235                     // name
236                     Method[] methods = klass.getMethods();
237                     for (int i = 0; methods != null && i < methods.length; i++) {
238                         Method method = methods[i];
239                         String methodName = method.getName();
240                         // see if the method is a setXXX
241                         if (methodName.startsWith("set")) {
242                             String propertyname = methodName.substring(3).toLowerCase();
243                             // found a setXXX method, so see if there is an XXX
244                             // property in the list read in from spy.properties.
245                             Enumeration keys = props.keys();
246                             while (keys.hasMoreElements()) {
247                                 String key = (String) keys.nextElement();
248                                 // all checks are all lower case
249                                 if (key.toLowerCase().equals(propertyname)) {
250                                     try {
251                                         // this is a parameter for the current method,
252                                         // so find out which supported type the method
253                                         // expects
254                                         String value = (String) props.get(key);
255                                         Class[] types = method.getParameterTypes();
256                                         if (types[0].getName().equals(value.getClass().getName())) {
257                                             // the method expects a string
258                                             String[] args = new String[1];
259                                             args[0] = value;
260                                             P6LogQuery.debug("calling " + methodName + " on DataSource " + rdsName + " with " + value);
261                                             method.invoke(rds, args);
262                                             matchedProps.put(key, value);
263                                         } else if (types[0].isPrimitive() && types[0].getName().equals("int")) {
264                                             // the method expects an int, so we pass an Integer
265                                             Integer[] args = new Integer[1];
266                                             args[0] = Integer.valueOf(value);
267                                             P6LogQuery.debug("calling " + methodName + " on DataSource " + rdsName + " with " + value);
268                                             method.invoke(rds, args);
269                                             matchedProps.put(key, value);
270                                         } else {
271                                             P6LogQuery.debug("method " + methodName + " on DataSource " + rdsName + " matches property "
272                                                 + propertyname + " but expects unsupported type " + types[0].getName());
273                                             matchedProps.put(key, value);
274                                         }
275                                     } catch (java.lang.IllegalAccessException e) {
276                                         throw (new SQLException("spy.properties file includes" + " datasource property " + key + " for datasource "
277                                             + rdsName + " but access is denied to method " + methodName));
278                                     } catch (java.lang.reflect.InvocationTargetException e2) {
279                                         throw (new SQLException("spy.properties file includes" + " datasource property " + key + " for datasource "
280                                             + rdsName + " but call method " + methodName + " fails"));
281                                     }
282                                 }
283                             }
284                         }
285                     }
286 
287                     Enumeration keys = props.keys();
288                     while (keys.hasMoreElements()) {
289 
290                         String key = (String) keys.nextElement();
291 
292                         if (!matchedProps.containsKey(key)) {
293                             P6LogQuery.debug("spy.properties file includes" + " datasource property " + key + " for datasource " + rdsName
294                                 + " but class " + klass.getName() + " has no method" + " by that name");
295                         }
296                     }
297                 }
298             }
299 
300         } catch (NamingException e) {
301             throw new SQLException("P6DataSource: naming exception during jndi lookup of Real Data Source Name of '" + rdsName + "'. "
302                 + e.getMessage());
303         }
304 
305         if (rds == null) {
306             throw new SQLException("P6DataSource: jndi lookup for Real Data Source Name of '" + rdsName + "' failed, cannot bind named data source.");
307         }
308     }

314     public Reference getReference() throws NamingException {
315         String FactoryName = "com.p6spy.engine.spy.P6DataSourceFactory";
316 
317         Reference Ref = new Reference(getClass().getName(), FactoryName, null);
318 
319         Ref.add(new StringRefAddr("dataSourceName", getRealDataSource()));
320         return Ref;
