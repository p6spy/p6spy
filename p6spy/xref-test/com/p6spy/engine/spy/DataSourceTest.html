157   class TestBasicDataSource extends BasicDataSource {
158     private boolean useDriverManager = true;
159 
160     void setUseDriverManager(final boolean useDriverManager) {
161       this.useDriverManager = useDriverManager;
162     }
163 
164     /**
165      * Creates a JDBC connection factory for this datasource.  This method only
166      * exists so subclasses can replace the implementation class.
167      */
168     @Override
169     protected ConnectionFactory createConnectionFactory() throws SQLException {
170       if (useDriverManager) {
171         return new ConnectionFactory() {
172           @Override
173           public Connection createConnection() throws SQLException {
174             return DriverManager.getConnection(getUrl(), getUsername(), getPassword());
175           }
176         };
177       } else {
178         return new ConnectionFactory() {
179           @Override
180           public Connection createConnection() throws SQLException {
181             Driver driver;
182             try {
183               driver = (Driver) P6Util.forName(getDriverClassName()).newInstance();
184             } catch (Exception e) {
185               throw new RuntimeException(e);
186             }
187             return new DriverConnectionFactory(driver, getUrl(), connectionProperties).createConnection();
188           }
189         };
190       }
191     }
192 
193     @Override
194     public Logger getParentLogger() throws SQLFeatureNotSupportedException {
195       throw new SQLFeatureNotSupportedException();
196     }
