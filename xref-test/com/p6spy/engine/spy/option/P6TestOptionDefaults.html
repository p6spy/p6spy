83      Assert.assertNull(opts.getDateformat());

91      Assert.assertNull(opts.getDriverNames());
92      Assert.assertFalse(opts.getStackTrace());
93      Assert.assertNull(opts.getStackTraceClass());
94      Assert.assertFalse(opts.getAutoflush());
95      Assert.assertFalse(opts.getReloadProperties());
96      Assert.assertEquals(60L, opts.getReloadPropertiesInterval());
97      Assert.assertNull(opts.getJNDIContextFactory());
98      Assert.assertNull(opts.getJNDIContextProviderURL());
99      Assert.assertNull(opts.getJNDIContextCustom());

149   @Test
150   public void testImplicitlyDisabledLoggedCategories() throws IOException {
151     {
152       final String msg = "debug logged test msg";
153       P6LogQuery.debug(msg);
154       final String logged = FileUtils.readFileToString(LOG_FILE, "UTF-8");
155       Assert.assertFalse(logged.contains(msg));
156     }
157     
158     {
159       final String msg = "info logged test msg";
160       P6LogQuery.info(msg);
161       final String logged = FileUtils.readFileToString(LOG_FILE, "UTF-8");
162       Assert.assertFalse(logged.contains(msg));
163     }
164     
165     {
166       final String msg = "error logged test msg";
167       P6LogQuery.error(msg);
168       final String logged = FileUtils.readFileToString(LOG_FILE, "UTF-8");
169       Assert.assertTrue(logged.contains(msg));
170     }
