106     @Before
107     public void setUpPreparedStatement() {
108         try {
109             Statement statement = connection.createStatement();
110             dropPrepared(statement);
111             statement.execute("create table prepstmt_test (col1 varchar(255), col2 integer)");
112         } catch (Exception e) {
113             fail(e.getMessage());
114         }
115     }
116 
117     @Test
118     public void testPreparedQueryUpdate() {
119         try {
120             // test a basic insert
121             String update = "insert into prepstmt_test values (?, ?)";
122             PreparedStatement prep = getPreparedStatement(update);
123             prep.setString(1, "miller");
124             prep.setInt(2,1);
125             prep.executeUpdate();
126             assertIsLastQuery(update);
127             assertIsLastQuery("miller");
128             assertIsLastQuery("1");
129 
130             // test a basic select
131             String query = "select count(*) from prepstmt_test where col2 = ?";
132             prep = getPreparedStatement(query);
133             prep.setInt(1,1);
134             ResultSet rs = prep.executeQuery();
135             assertIsLastQuery(query);
136             rs.next();
137             assertEquals(1, rs.getInt(1));
138 
139             // test dynamic allocation of P6_MAX_FIELDS
140             int MaxFields = P6PreparedStatement.P6_MAX_FIELDS + 3;
141             StringBuffer bigSelect = new StringBuffer(MaxFields);
142             bigSelect.append("select count(*) from prepstmt_test where");
143             for (int i = 0; i < MaxFields; i++) {
144                 if (i > 0) {
145                   bigSelect.append(" or ");
146                 }
147                 bigSelect.append(" col2=?");
148             }
149             prep = getPreparedStatement(bigSelect.toString());
150             for (int i = 1; i <= MaxFields; i++) {
151                  prep.setInt(i, i);
152 	    }
153             //rs = prep.executeQuery();
154 
155             // test batch inserts
156             update = "insert into prepstmt_test values (?,?)";
157             prep = getPreparedStatement(update);
158             prep.setString(1,"danny");
159             prep.setInt(2,2);
160             prep.addBatch();
161             prep.setString(1,"denver");
162             prep.setInt(2,3);
163             prep.addBatch();
164             prep.setString(1,"aspen");
165             prep.setInt(2,4);
166             prep.addBatch();
167             prep.executeBatch();
168             assertIsLastQuery(update);
169             assertIsLastQuery("aspen");
170             assertIsLastQuery("4");
171 
172             query = "select count(*) from prepstmt_test";
173             prep = getPreparedStatement(query);
174             rs = prep.executeQuery();
175             rs.next();
176             assertEquals(4, rs.getInt(1));
177         } catch (Exception e) {
178             fail(e.getMessage()+" due to error: "+getStackTrace(e));
179         }
180     }
181 
182     @After
183     public void tearDownPreparedStatement() {
184         try {
185             Statement statement = connection.createStatement();
186             dropPrepared(statement);
187         }  catch (Exception e) {
188             fail(e.getMessage());
189         }
190     }
191 
192     protected void dropPrepared(Statement statement) {
193         dropPreparedStatement("drop table prepstmt_test", statement);
194     }
195 
196     protected void dropPreparedStatement(String sql, Statement statement) {
197         try {
198             statement.execute(sql);
199         } catch (Exception e) {
200             // we don't really care about cleanup failing
201         }
202     }
203 
204     protected PreparedStatement getPreparedStatement(String query) throws SQLException {
205         return (connection.prepareStatement(query));
206     }
