49      public SubclassTask() {
50          fileSets = new ArrayList<FileSet>();
51      }

53      @Override
54      public void execute() throws BuildException {
55          // use the fileset defined in the parent (abstract) class
56          loader = createClassLoader();
57  
58          for (FileSet fileset : fileSets) {
59              DirectoryScanner ds = fileset.getDirectoryScanner(getProject());
60              String[] files = ds.getIncludedFiles();
61  
62              if ( files != null) {
63                  for (String fileName: files) {
64                      if (fileName.endsWith(".class")) {
65                          fixAndLoad(fileName);
66                      } else if (fileName.endsWith(".jar")) {
67                          expandJar(ds.getBasedir(), fileName);
68                      } else {
69                          log("File '" + fileName + "' is neither a .class nor a .jar file, skipping.", Project.MSG_WARN);
70                      }
71                  }
72              }
73          }
74      }
75      public void fixAndLoad(String fileName) {
76          // strip off the ".class" at the end
77          String cleanString = fileName.substring(0, fileName.indexOf(".class"));
78          StringBuffer className = new StringBuffer();
79  
80          // change all of the delims to dots... don't use the
81          // replace methods on String to preserve some backwards compat
82          int start = 0;
83          int pos = 0;
84          while ((pos = cleanString.indexOf(Subclasser.DELIMITER, start)) > 0) {
85              className.append(cleanString.substring(start, pos));
86              className.append(".");
87              start = pos + 1;
88          }
89  
90          // catch the tail
91          if (start < cleanString.length()) {
92              className.append(cleanString.substring(start));
93          }
94  
95          cleanString = className.toString();
96          // try not to further subclass P6 Stuff
97          // this is a little weak, but whatever
98          if (cleanString.indexOf(".P6") == -1) {
99              loadClass(cleanString);
100         }
101     }

103     public void loadClass(String className) {
104         try {
105             Class k = loader.loadClass(className);
106             int mods = k.getModifiers();
107 
108             // make sure it's a DataSource and something
109             // instantiatable (whew)
110             if (DataSource.class.isAssignableFrom(k) && ((mods & Modifier.ABSTRACT) == 0) && ((mods & Modifier.INTERFACE) == 0)
111                 && ((mods & Modifier.FINAL) == 0)) {
112 
113                 log("Found class " + className, Project.MSG_WARN);
114 
115                 Subclasser sub = new Subclasser(k);
116 
117                 if (outputPackage != null) {
118                     sub.setOutputPackage(outputPackage);
119                 }
120 
121                 if (outputDir != null) {
122                     sub.setOutputDir(outputDir);
123                 }
124 
125                 sub.createSubClass();
126             }
127         } catch (Exception e) {
128             throw new BuildException(e);
129         }
130     }
131 
132     public void expandJar(File parentDir, String jarName) {
133         try {
134             File jarFS = new File(parentDir, jarName);
135             JarFile jar = new JarFile(jarFS);
136 
137             for (Enumeration e = jar.entries(); e.hasMoreElements();) {
138                 JarEntry entry = (JarEntry) e.nextElement();
139 
140                 if (!entry.isDirectory() && entry.getName().endsWith(".class")) {
141                     fixAndLoad(entry.getName());
142                 }
143 
144             }
145         } catch (IOException e) {
146             throw new BuildException(e);
147         }
148 
149     }
