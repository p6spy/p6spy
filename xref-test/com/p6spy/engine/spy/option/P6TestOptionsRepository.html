18  import java.util.Arrays;
19  import java.util.HashSet;
20  import java.util.Set;
21  import java.util.regex.Pattern;

23  import org.junit.Assert;
24  import org.junit.Before;
25  import org.junit.Test;
26  
27  import com.p6spy.engine.spy.appender.MessageFormattingStrategy;
28  import com.p6spy.engine.spy.appender.MultiLineFormat;
29  import com.p6spy.engine.spy.appender.SingleLineFormat;

31  public class P6TestOptionsRepository {
32  
33    private P6OptionsRepository optRepo;
34  
35    @Before
36    public void setUp() {
37      optRepo = new P6OptionsRepository();
38    }
39  
40    @Test
41    public void testParse() {
42      Assert.assertEquals("", optRepo.parse(String.class, ""));
43      Assert.assertEquals(100, optRepo.parse(Integer.class, 100));
44      Assert.assertEquals(100L, optRepo.parse(Long.class, 100L));
45      Assert.assertEquals(true, optRepo.parse(Boolean.class, "true"));
46      Assert.assertTrue(optRepo.parse(MessageFormattingStrategy.class,
47          SingleLineFormat.class.getName()) instanceof MessageFormattingStrategy);
48      Assert.assertTrue(optRepo.parse(MessageFormattingStrategy.class,
49          MultiLineFormat.class.getName()) instanceof MessageFormattingStrategy);
50      Assert.assertTrue(optRepo.parse(Pattern.class,
51          "somepattern") instanceof Pattern);
52    }
53  
54    @Test(expected = IllegalArgumentException.class)
55    public void testParseForCollectionFails() {
56      optRepo.parse(Set.class, new HashSet<String>());
57      Assert.fail();
58    }
59  
60    @Test(expected = IllegalArgumentException.class)
61    public void testParseForListFails() {
62      optRepo.parse(Set.class, new HashSet<String>());
63      Assert.fail();
64    }
65  
66    @Test(expected = IllegalArgumentException.class)
67    public void testParseForSetFails() {
68      optRepo.parse(Set.class, new HashSet<String>());
69      Assert.fail();
70    }
71  
72    @Test(expected = IllegalStateException.class)
73    public void testGetPrioToInitCompletedFails() {
74      optRepo.set(String.class, "option1", "");
75      Assert.assertEquals("", optRepo.get(String.class, "option1"));
76    }
77  
78    @Test
79    public void testSetGetSimple() {
80      optRepo.initCompleted();
81  
82      optRepo.set(String.class, "option1", "value1");
83      Assert.assertEquals("value1", optRepo.get(String.class, "option1"));
84    }
85  
86    @Test
87    public void testSetSetAdditionAndRemoval() {
88      optRepo.initCompleted();
89  
90      optRepo.setSet(String.class, "option1", "value1,value2");
91      Assert.assertEquals(new HashSet<String>(Arrays.asList("value1", "value2")),
92          optRepo.get(String.class, "option1"));
93  
94      optRepo.setSet(String.class, "option1", "value3,-value2");
95      Assert.assertEquals(new HashSet<String>(Arrays.asList("value1", "value3")),
96          optRepo.get(String.class, "option1"));
97    }
