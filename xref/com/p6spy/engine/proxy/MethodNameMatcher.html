16  package com.p6spy.engine.proxy;
17  
18  import java.lang.reflect.Method;
19  
20  /**
21   * @author Quinton McCombs
22   * @since 09/2013
23   */
24  public class MethodNameMatcher implements MethodMatcher {
25    private final String methodName;
26    private final boolean startsWith;
27  
28    public MethodNameMatcher(final String methodName) {
29      if( methodName.endsWith("*") ) {
30        this.startsWith = true;
31        this.methodName = methodName.substring(0, methodName.length() - 1);
32      } else {
33        this.startsWith = false;
34        this.methodName = methodName;
35      }
36  
37      // wildcard is only allowed at end of method name
38      if( this.methodName.contains("*")) {
39        throw new IllegalArgumentException("Wildcard is only allowed at end of method name");
40      }
41    }
42  
43    @Override
44    public boolean matches(final Method method) {
45      boolean result;
46      if( startsWith ) {
47        result = method.getName().startsWith(methodName);
48      } else {
49        result = method.getName().equals(methodName);
50      }
51      return result;
52    }
53  
54    @Override
55    public boolean equals(final Object o) {
56      if (this == o) return true;
57      if (!(o instanceof MethodNameMatcher)) return false;
58  
59      MethodNameMatcher that = (MethodNameMatcher) o;
60  
61      if (startsWith != that.startsWith) return false;
62      if (!methodName.equals(that.methodName)) return false;
63  
64      return true;
65    }
66  
67    @Override
68    public int hashCode() {
69      int result = methodName.hashCode();
70      result = 31 * result + (startsWith ? 1 : 0);
71      return result;
72    }
73  }
