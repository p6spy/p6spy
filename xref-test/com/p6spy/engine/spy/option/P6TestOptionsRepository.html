46    @Test
47    public void testParse() {
48      Assert.assertEquals("", optRepo.parse(String.class, ""));
49      Assert.assertEquals(100, optRepo.parse(Integer.class, 100));
50      Assert.assertEquals(100L, optRepo.parse(Long.class, 100L));
51      Assert.assertEquals(true, optRepo.parse(Boolean.class, "true"));
52      Assert.assertTrue(optRepo.parse(MessageFormattingStrategy.class,
53          SingleLineFormat.class.getName()) instanceof MessageFormattingStrategy);
54      Assert.assertTrue(optRepo.parse(MessageFormattingStrategy.class,
55          MultiLineFormat.class.getName()) instanceof MessageFormattingStrategy);
56      Assert.assertTrue(optRepo.parse(Pattern.class,
57          "somepattern") instanceof Pattern);
58  
59      // existing categories work
60      Assert.assertTrue(optRepo.parse(Category.class,
61              "info") instanceof Category);
62      // new categories can be added without restriction
63      Assert.assertTrue(optRepo.parse(Category.class,
64              "new_category") instanceof Category);
65    }
