62  package com.p6spy.engine.logging.appender;
63  
64  import org.apache.log4j.*;
65  import java.io.StringWriter;
66  import java.io.PrintWriter;
67  
68  import com.p6spy.engine.common.*;
69  
70  // adapted from Rafael Alvarez's LoggingStream class
71  public class Log4jLogger extends FormattedLogger implements P6Logger {
72      
73      protected Level level = Level.INFO;
74      protected String lastEntry;
75      private static Logger log;
76      
77      // By configuring log4j by this method, we control the p6spy logger behavior
78      // using the same configuration file as p6spy, and any other attempt
79      // to configure log4j will add to this configuration.
80      public Log4jLogger() {
81          P6SpyProperties properties = new P6SpyProperties();
82          PropertyConfigurator.configure(properties.forceReadProperties());
83          log = Logger.getLogger("p6spy");
84          log.setAdditivity(false);
85      }
86      
87      
88      public void logException(Exception e) {
89          StringWriter sw = new StringWriter();
90          PrintWriter pw  = new PrintWriter(sw);
91          e.printStackTrace(pw);
92          logText(sw.toString());
93      }
94      
95      public void logText(String text) {
96          log.log(this.level, text);
97          setLastEntry(text);
98      }
99      
100     public Level getLevel() {
101         return this.level;
102     }
103     
104     public void setLevel(Level inVar) {
105         this.level = inVar;
106     }   
107 }
