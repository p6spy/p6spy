24  import com.p6spy.engine.proxy.GenericInvocationHandler;

27  import java.sql.Statement;

30  /**
31   * Invocation handler for {@link java.sql.Statement}
32   */
33  class P6LeakStatementInvocationHandler extends GenericInvocationHandler<Statement> {
34  
35    public P6LeakStatementInvocationHandler(Statement underlying, final ConnectionInformation connectionInformation) {
36      super(underlying);
37      StatementInformation statementInformation = new StatementInformation(connectionInformation);
38      P6Objects.open(statementInformation);
39  
40      P6LeakStatementCloseDelegate closeDelegate = new P6LeakStatementCloseDelegate(statementInformation);
41      P6LeakStatementExecuteDelegate executeDelegate = new P6LeakStatementExecuteDelegate(statementInformation);
42  
43      addDelegate(
44          new MethodNameMatcher("close"),
45          closeDelegate
46      );
47      addDelegate(
48          new MethodNameMatcher("execute*"),
49          executeDelegate
50      );
51  
52    }
53  
54  }
