137     /**
138      * This gets overloaded in the P6LogResultSet, but may need to do what that class does
139      */
140     public boolean next() throws SQLException {
141         // only dump the data on subsequent calls to next
142         if (currRow > -1) {
143             StringBuilder buffer = new StringBuilder();
144             String comma = "";
145             for (Map.Entry<String, Object> entry : resultMap.entrySet()) {
146                 buffer.append(comma);
147                 buffer.append(entry.getKey());
148                 buffer.append(" = ");
149                 buffer.append(entry.getValue());
150                 comma = ", ";
151             }
152             P6LogQuery.log("resultset", query, buffer.toString());
153             resultMap.clear();
154         }
155         currRow++;
156         return passthru.next();
157     }
