39  @RunWith(Parameterized.class)

41    private static final Logger log = Logger.getLogger(P6TestCallableStatement.class);

43    @Parameterized.Parameters(name = "{index}: {0}")
44    public static Collection<Object[]> dbs() {
45      Collection<Object[]> result;
46      String dbList = (System.getProperty("DB") == null ? "HSQLDB" : System.getProperty("DB"));
47  
48      Object[] dbs = dbList.split(",");
49      List<Object[]> dbsToTest = new ArrayList<Object[]>();
50      for (int i = 0; i < dbs.length; i++) {
51        //  Check against list of databases with stored procs
52        // As procs become available for other databases, enable them here.
53        if( Arrays.asList("Oracle","MySQL","HSQLDB").contains(dbs[i])) {
54          dbsToTest.add(new Object[]{dbs[i]});
55        } else {
56          log.info("Skipping "+dbs[i]+" because stored procedures have not been created for testing");
57        }
58      }
59      result = dbsToTest;
60  
61      return result;
62    }

65    public P6TestCallableStatement(String db) throws SQLException, IOException {
66      super(db);
67    }

69    @Test
70    public void testStoredProcedureNoResultSet() throws SQLException {
71      this.clearLogEnties();
72  
73      // execute the statement
74      String query = "{call test_proc(?,?,?)}";
75      CallableStatement call = connection.prepareCall(query);
76      call.registerOutParameter(3, Types.INTEGER);
77      call.setInt(1, 1);
78      call.setString(2, "hi");
79      call.execute();
80      int retVal = call.getInt(3);
81      assertEquals(2, retVal);
82      call.close();
83  
84      // the last log message should have the original query
85      assertTrue(getLastLogEntry().contains(query));
86  
87      // verify that the bind parameters are resolved in the log message
88      assertTrue(getLastLogEntry().contains("1,'hi'"));
89    }

91    @Test
92    public void testNamedParameters() throws SQLException {
93      this.clearLogEnties();
94      
95      String param1Name = "param1";
96      String param2Name = "param2";
97      String resultParamName = "result_param";
98      
99      if( "HSQLDB".equals(db) ) {
100       // HSQLDB uses @p1, @p2, etc...  as the "names" of the parameters
101       param1Name = "@p1";
102       param2Name = "@p2";
103       resultParamName = "@p3";
104     }
105 
106     // execute the statement
107     String query = "{call test_proc(?,?,?)}";
108     CallableStatement call = connection.prepareCall(query);
109     call.setInt(param1Name, 1);
110     call.setString(param2Name, "hi");
111     call.registerOutParameter(resultParamName, Types.INTEGER);
112     call.execute();
113     int retVal = call.getInt(3);
114     assertEquals(2, retVal);
115     call.close();
116 
117     // the last log message should have the original query
118     assertTrue(getLastLogEntry().contains(query));
119 
120     assertTrue(getLastLogEntry().contains("{call test_proc(?,?,?)}"));
121     assertTrue(getLastLogEntry().contains(param1Name+":"+"1"));
122     assertTrue(getLastLogEntry().contains(param2Name+":"+"'hi'"));
123   }
