65    @Test
66    public void testJmxSetPropertyDiscartedOnExplicitJmxReload() throws Exception {
67      final String domainName = P6SpyOptions.class.getPackage().getName();
68      final String beanName = P6SpyOptions.class.getSimpleName();
69      final String attributeName = "StackTrace";
70  
71      // precondition
72      assertFalse((Boolean) jmxClient.getAttribute(domainName, beanName, attributeName));
73  
74      // jmx value modification
75      jmxClient.setAttribute(domainName, beanName, attributeName, true);
76      assertTrue((Boolean) jmxClient.getAttribute(domainName, beanName, attributeName));
77  
78      // props reload
79      jmxClient.invokeOperation(P6SpyOptions.class.getPackage().getName(),
80          P6SpyOptions.class.getSimpleName(), "reload");
81  
82      // jmx value modification discarted
83      assertFalse((Boolean) jmxClient.getAttribute(domainName, beanName, attributeName));
84    }

241   @Test
242   public void testSpyDotPropertiesWithSpaceInPathWorks() throws Exception {
243     //
244     // setup preconditions
245     //
246     final File spyDotPropertiesWithSpaceInPath;
247     // create spy.properties file with the space in path
248     {
249       final File tmpFile = File.createTempFile("whatever", ".tmp");
250       final File tmpDir = tmpFile.getParentFile();
251       tmpFile.delete();
252       final File dirWithSpace = new File(tmpDir.getAbsolutePath() + "/path with space");
253       dirWithSpace.deleteOnExit();
254       dirWithSpace.mkdir();
255       assertTrue(dirWithSpace.exists());
256       final File source = new File(P6TestFramework.TEST_FILE_PATH, "P6Test_reload_2.properties");
257       spyDotPropertiesWithSpaceInPath = new File(dirWithSpace, "P6Test_reload_2.properties");
258       FileUtils.copyFile(source, spyDotPropertiesWithSpaceInPath);
259     }
260 
261     //
262     // full path in system property case
263     //
264 
265     // ensure property loaded from file correctly
266     // [default] stacktrace=false
267     // [SpyDotProperties] stacktrace=true
268     // => true
269     System.setProperty(SpyDotProperties.OPTIONS_FILE_PROPERTY,
270         spyDotPropertiesWithSpaceInPath.getAbsolutePath());
271     P6SpyOptions.getActiveInstance().reload();
272 
273     assertTrue(P6SpyOptions.getActiveInstance().getStackTrace());
274 
275     //
276     // relative path with p6.home in system properties case
277     //
278 
279     // ensure property loaded from file correctly
280     // [default] stacktrace=false
281     // [SpyDotProperties] stacktrace=true
282     // => true
283     System.setProperty(SpyDotProperties.OPTIONS_FILE_PROPERTY,
284         spyDotPropertiesWithSpaceInPath.getName());
285     System
286         .setProperty("p6.home", spyDotPropertiesWithSpaceInPath.getParentFile().getAbsolutePath());
287     P6SpyOptions.getActiveInstance().reload();
288 
289     assertTrue(P6SpyOptions.getActiveInstance().getStackTrace());
290   }
