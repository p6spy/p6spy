64    protected void configure(String log4jConfSuffix, boolean removeDefaultExcludedCategories) throws Exception {
65  
66  	if (removeDefaultExcludedCategories) {
67  		// we test slf4j filtering here rather than categories one
68  		P6LogOptions.getActiveInstance().setExcludecategories("");
69  	}
70  	
71      // reset log4j
72      LogManager.resetConfiguration();
73  
74      // configure log4j externally
75      configureLog4JInTest(log4jConfSuffix);
76    }
77  
78    @After
79    public void cleanup() throws Exception {
80  	// restore default excluded categories
81  	P6LogOptions.getActiveInstance().setExcludecategories(
82  			StringUtils.join(P6TestOptionDefaults.DEFAULT_CATEGORIES, ","));
83  	
84      framework.closeConnection();
85      
86      // reset log4j
87      LogManager.resetConfiguration();
88  
89      // load default configuration
90      configureLog4J();
91    }

156   public void checkCategoryToLevelMapping(Category category, Level expectedLevel, Level thresholdLevel) throws Exception {
157   		configure(thresholdLevel.toString(), true);
158 
159 		Log4JTestApppender.clearCapturedMessages();
160 	    P6LogQuery.log(category, "sample msg", "sample msg");
161 	    
162 	    if (expectedLevel == null) {
163 	    	assertEquals(0, Log4JTestApppender.getCapturedMessages().size());
164 	    } else {
165 	    	assertEquals(1, Log4JTestApppender.getCapturedMessages().size());
166 		    assertEquals(expectedLevel, Log4JTestApppender.getCapturedMessages().get(0).getLevel());	
167 	    }
