287      * P6Spy specific functionality
288      */
289     public final String getQueryFromPreparedStatement() {
290         int len = preparedQuery.length();
291         StringBuffer t = new StringBuffer(len * 2);
292 
293         if (values != null) {
294             int i = 1, limit = 0, base = 0;
295 
296             while ((limit = preparedQuery.indexOf('?', limit)) != -1) {
297                 t.append(preparedQuery.substring(base, limit));
298                 if (values[i] == null) {
299                     t.append("NULL");
300                 } else if (isString[i]) {
301                     t.append("'");
302                     t.append(values[i]);
303                     t.append("'");
304                 } else {
305                     t.append(values[i]);
306                 }
307                 i++;
308                 limit++;
309                 base = limit;
310             }
311             if (base < len) {
312                 t.append(preparedQuery.substring(base));
313             }
314         }
315 
316         return t.toString();
317     }

319     protected void growValues(int newMax) {
320         int size = values.length;
321         Object[] values_tmp = new Object[newMax + P6_GROW_MAX];
322         boolean[] isString_tmp = new boolean[newMax + P6_GROW_MAX];
323         System.arraycopy(values, 0, values_tmp, 0, size);
324         values = values_tmp;
325         System.arraycopy(isString, 0, isString_tmp, 0, size);
326         isString = isString_tmp;
327     }
328 
329     protected void setObjectAsString(int i, Object o) {
330         if (values != null) {
331             if (i >= 0) {
332                 if (i >= values.length) {
333                     growValues(i);
334                 }
335                 if (o instanceof java.util.Date) {
336                     values[i] = new SimpleDateFormat(P6SpyOptions.getDatabaseDialectDateFormat()).format(o);
337                 }
338                 else if ( o instanceof byte[] ) {
339                     values[i] = toHexString( (byte[])o );
340                 }
341                 else {
342                     values[i] = (o == null) ? null : o.toString();
343                 }
344                 isString[i] = true;
345             }
346         }
347     }
348 
349     private static final String toHexString( byte[] bytes ) {
350         String value = "";
351         for ( byte b : bytes ) {
352             int temp = (int)b & 0xFF;
353             value += HEX_CHARS[ temp/16 ];
354             value += HEX_CHARS[ temp%16 ];
355         }
356         return value;
357     }

359     protected void setObjectAsInt(int i, Object o) {
360         if (values != null) {
361             if (i >= 0) {
362                 if (i >= values.length) {
363                     growValues(i);
364                 }
365                 values[i] = (o == null) ? null : o.toString();
366                 isString[i] = false;
367             }
368         }
369     }
